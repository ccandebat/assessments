function downloadPDF() { const btn = document.getElementById('dlBtn'); btn.disabled = true; btn.textContent = 'Generating…'; // Hide the toolbar const toolbar = document.getElementById('toolbar'); toolbar.style.display = 'none'; // Temporarily fix styles that break html2pdf rendering: // - overflow:hidden clips content beyond viewport height // - margin creates blank space at top of canvas // - border-radius can cause rendering artifacts const wrapper = document.getElementById('formContent'); const origOverflow = wrapper.style.overflow; const origMargin = wrapper.style.margin; const origRadius = wrapper.style.borderRadius; const origShadow = wrapper.style.boxShadow; wrapper.style.overflow = 'visible'; wrapper.style.margin = '0 auto'; wrapper.style.borderRadius = '0'; wrapper.style.boxShadow = 'none'; const opt = { margin: [0.3, 0.4, 0.3, 0.4], filename: 'Initial_Triage_Checklist.pdf', image: { type: 'jpeg', quality: 0.98 }, html2canvas: { scale: 2, useCORS: true, logging: false }, jsPDF: { unit: 'in', format: 'letter', orientation: 'portrait' }, pagebreak: { mode: ['css'], avoid: ['.decision-box', '.sig-block', '.invention-info', '.pass-fail'] } }; html2pdf().set(opt).from(wrapper).save().then(() => { // Restore original styles wrapper.style.overflow = origOverflow; wrapper.style.margin = origMargin; wrapper.style.borderRadius = origRadius; wrapper.style.boxShadow = origShadow; toolbar.style.display = ''; btn.disabled = false; btn.textContent = 'Download as PDF'; }); }
